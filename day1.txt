import React, { useEffect, useState } from "react";
import { useNavigate } from "react-router-dom";
function Register(props) {
  const navigate = useNavigate();
  const [users, setUsers] = useState([]);
  const handleChange = (e) => {
    setUsers({ ...users, [e.target.name]: e.target.value });
  };
  const addUser = (e) => {
    e.preventDefault();

    localStorage.setItem("udata", JSON.stringify(users));
    alert("User Registerd");
    navigate("/login");
  };
  return (
    <div>
      <form onSubmit={addUser}>
        UserName:
        <input type="text" name="uname" onChange={handleChange} />
        Password:
        <input type="password" name="password" onChange={handleChange} />
        <button>Add User</button>
        <br />
      </form>
    </div>
  );
}

export default Register;

import React, { useEffect, useState } from "react";
import { useNavigate } from "react-router-dom";

function Login(props) {
  const navigate = useNavigate();
  const [users, setUsers] = useState([]);
  const handleChange = (e) => {
    setUsers({ ...users, [e.target.name]: e.target.value });
  };
  const login = (e) => {
    e.preventDefault();
    let validate = JSON.parse(localStorage.getItem("udata"));
    if (validate.uname == users.uname && validate.password == users.password) {
      navigate("/userdetails");
    }else{
        alert('Incorrect Credentials')
    }
  };
  return (
    <div>
      <form onSubmit={login}>
        UserName:
        <input type="text" name="uname" onChange={handleChange} />
        Password:
        <input type="password" name="password" onChange={handleChange} />
        <button>Add User</button>
        <br />
      </form>
    </div>
  );
}

export default Login;
